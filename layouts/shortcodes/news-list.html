{{- /* 从 news.md 读取新闻并显示 */ -}} {{- /* 接受 limit 参数，默认为 5 */ -}} {{- $limit := .Get "limit" | default "5" -}} {{- $newsContent := "" -}} {{- /* 使用 Hugo 页面系统读取 */ -}} {{- $newsPage := site.GetPage "/post/news" -}} {{- if $newsPage -}} {{- $newsContent = $newsPage.RawContent -}} {{- end -}} {{- if $newsContent -}} {{- /* 移除 front matter 分隔符 */ -}} {{- $content := $newsContent -}} {{- if hasPrefix $content "---" -}} {{- $parts := split $content "---" -}} {{- if ge (len $parts) 3 -}} {{- $content = index $parts 2 -}} {{- end -}} {{- end -}} {{- $content = trim $content " \n\r" -}} {{- $lines := split $content "\n" -}} {{- $newsItems := slice -}} {{- range $lines -}} {{- $line := trim . " \r\n\t" -}} {{- if and $line (hasPrefix $line "[") -}} {{- /* 使用 findRE 提取日期和标题 */ -}} {{- $matches := findRE `\[([^\]]+)\]\s*(.+)` $line 1 -}} {{- if $matches -}} {{- $match := index $matches 0 -}} {{- /* 分别提取日期和标题部分 */ -}} {{- $dateMatch := findRE `\[([^\]]+)\]` $match 1 -}} {{- $date := "" -}} {{- if $dateMatch -}} {{- $date = replaceRE `\[|\]` "" (index $dateMatch 0) -}} {{- end -}} {{- $title := replaceRE `^\[[^\]]+\]\s*` "" $match -}} {{- if and $date $title -}} {{- $newsItems = $newsItems | append (dict "date" $date "title" $title) -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
{{- if eq $limit "all" -}} {{- range $newsItems -}}
[{{ .date }}] {{ .title }}
{{- end -}} {{- else -}} {{- range first (int $limit) $newsItems -}}
[{{ .date }}] {{ .title }}
{{- end -}} {{- end -}}
{{- else -}}

No news available.

{{- end -}}
